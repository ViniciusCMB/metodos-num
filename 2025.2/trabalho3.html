def resolver_difusao(t_total, n, L, alpha, dt, T_A, q_L, T_inicial):
    """
    Resolve a equao de difuso de calor 1D (EDP parablica) usando o mtodo
    de diferenas finitas totalmente implcito.
    """
    
    # 1. Discretizao
    dx = L / (n - 1)
    Nt = int(t_total / dt)
    
    # 2. Inicializao da Matriz A
    A = np.zeros((n, n))
    
    # 3. Preenchimento da Matriz A (ns internos)
    # (-alpha/dx^2)T_{i-1}^{k+1} + (1/dt + 2*alpha/dx^2)T_i^{k+1} + (-alpha/dx^2)T_{i+1}^{k+1} = T_i^k / dt
    for i in range(1, n - 1):
        A[i, i-1] = -alpha / dx**2
        A[i, i]   = 1 / dt + 2 * alpha / dx**2
        A[i, i+1] = -alpha / dx**2

    # 4. Condies de Contorno na Matriz A
    A[0, 0] = 1.0     # Dirichlet em x=0
    A[n-1, n-1] = 1.0 # Neumann em x=L (dT/dx = q_L)
    A[n-1, n-2] = -1.0

    # 5. Condio Inicial
    T = np.full(n, T_inicial)
    
    # 6. Armazenamento (Apenas o ltimo passo  necessrio para este estudo)
    # Loop no Tempo
    for k in range(Nt):
        b = T / dt
        b[0] = T_A
        b[n-1] = q_L * dx
        
        T = np.linalg.solve(A, b)

    # 7. Cria o vetor de posies x
    x = np.linspace(0, L, n)
    
    # Retorna o ltimo perfil de T e o vetor x
    return T, x
